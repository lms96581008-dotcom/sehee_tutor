/* ═══ RENDER CALENDAR ═══ */

function renderCal(){

  const y=calM.getFullYear(),m=calM.getMonth();

  document.getElementById('cal-title').textContent=`${y}년 ${MO[m]}`;

  document.getElementById('sb-month').textContent=MO[m];

  const first=new Date(y,m,1).getDay();

  const dim=new Date(y,m+1,0).getDate();

  const total=Math.ceil((first+dim)/7)*7;

  let html='';

  for(let i=0;i<total;i++){

    let dn,ds,om=false;

    const col=i%7;

    if(i<first){dn=new Date(y,m,0).getDate()-first+i+1;const pm=m===0?12:m,py=m===0?y-1:y;ds=`${py}-${String(pm).padStart(2,'0')}-${String(dn).padStart(2,'0')}`;om=true;}

    else if(i<first+dim){dn=i-first+1;ds=`${y}-${String(m+1).padStart(2,'0')}-${String(dn).padStart(2,'0')}`;}

    else{dn=i-first-dim+1;const nm=m===11?1:m+2,ny=m===11?y+1:y;ds=`${ny}-${String(nm).padStart(2,'0')}-${String(dn).padStart(2,'0')}`;om=true;}

    const isT=!om&&ds===todayStr;

    // Events for this day (cal + notes with date)

    const cev=CAL[ds]||[];

    const nev=S.flatMap(s=>(N[s.id]||[]).filter(n=>n.date===ds).map(n=>({...n,_student:s,fromNote:true})));

    const allEv=[...cev,...nev];

    const hasE=allEv.length>0;

    html+=`<div class="cday${isT?' today':''}${om?' other':''}${hasE?' hase':''}" onclick="openEv('${ds}')">

      <div class="cdnum" style="${!isT&&col===0?'color:#EF4444':!isT&&col===6?'color:#3B82F6':''}">${dn}</div>

      <div class="cevents">

        ${allEv.slice(0,2).map(e=>{const c=CATS[e.category]||CATS.general;return`<div class="cetag" style="background:${c.bg};color:${c.c};border:1px solid ${c.bd}">${c.i} ${esc((e.text||'').slice(0,9))}${(e.text||'').length>9?'…':''}</div>`;}).join('')}

        ${allEv.length>2?`<div class="cmore">+${allEv.length-2}개</div>`:''}

      </div>

    </div>`;

  }

  document.getElementById('cal-grid').innerHTML=html;

  renderSB();

}


function renderSB(){

  const y=calM.getFullYear(),m=calM.getMonth();

  const evs=[];

  for(const [ds,arr] of Object.entries(CAL)){

    const d=new Date(ds+'T00:00:00');

    if(d.getFullYear()===y&&d.getMonth()===m) arr.forEach(e=>evs.push({...e,date:ds,d}));

  }

  S.forEach(s=>(N[s.id]||[]).forEach(n=>{

    if(n.date){const d=new Date(n.date+'T00:00:00');if(d.getFullYear()===y&&d.getMonth()===m)evs.push({...n,studentId:s.id,date:n.date,d,fromNote:true});}

  }));

  evs.sort((a,b)=>a.d-b.d);

  if(!evs.length){document.getElementById('sb-events').innerHTML=`<div class="sb-empty">이 달에 일정이 없습니다<br><small>날짜를 클릭하면 추가할 수 있어요</small></div>`;return;}

  document.getElementById('sb-events').innerHTML=evs.map(e=>{

    const c=CATS[e.category]||CATS.general;

    const st=e.studentId?S.find(s=>s.id===e.studentId):null;

    return`<div class="ritem" style="border-left-color:${c.c}">

      <span style="font-size:14px;flex-shrink:0">${c.i}</span>

      <div style="flex:1;min-width:0">

        <div class="ri-text">${esc(e.text)}</div>

        ${st?`<div class="ri-stu">${esc(st.kr)} · ${esc(st.en)}</div>`:''}

        <div class="ri-date">📅 ${e.date}</div>

      </div>

      ${!e.fromNote?`<button onclick="delEv('${e.date}','${e.id}')" style="background:none;border:none;color:#CBD5E1;cursor:pointer;font-size:13px;flex-shrink:0">✕</button>`:''}

    </div>`;

  }).join('');

}
